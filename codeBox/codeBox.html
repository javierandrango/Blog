                void blinkLed(int time_ms){
                  /**
                   * n: number of times the led will blink
                   * time_ms: time perior between blinks
                  */
                  current_time = millis();
                  if (current_time - previous_time>=time_ms){
                    //change state
                    if (led_state == LOW){
                      led_state = HIGH;
                    }
                    else{
                      led_state = LOW;
                    }
                    //update time
                    previous_time = current_time;
                  }
                  digitalWrite(BUILTIN_LED, led_state);
                }
